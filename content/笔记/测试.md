        transition: opacity 0.3s; /* 添加过渡效果 */
    }

    /* 当鼠标悬停在上面时显示提示框 */
    .tooltip:hover .tooltiptext {
        visibility: visible; /* 显示提示框 */
        opacity: 1; /* 渐变显示 */
    }
